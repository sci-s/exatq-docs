132 ErrorType *curr_error = nullptr;

137 }

139 // Computing Blocks
140 printf("\t... Computing reconstruction\n");
141
142 std::thread producer([&]() {
143 for (uint64_t iter = 0; iter < tasks.size(); ++iter) {
144 uint64_t block_id = tasks[iter].block_id;
145 manager->ReadBlockFromFile(curr_block, block_id);
146 curr_error = error_T[block_id];
147
148 index_t *curr_block_coord = curr_block->get_block_coord();
149 for (int axis = 0; axis < tensor->order; ++axis) {
150 if (factor_matrices[axis][curr_block_coord[axis]] == nullptr) {
151 factor_matrices[axis][curr_block_coord[axis]] = cputucker::allocate<value_t>(tensor->block_dims[axis] * rank);
152 manager->ReadFactorMatrixFromFile(factor_matrices[axis][curr_block_coord[axis]], tensor, rank, axis, curr_block_coord[axis]);
153 }
154 ++factor_matrices_ref_counter[axis][curr_block_coord[axis]];
155 }
156
157 task_t task = {curr_block, curr_error};
158
159 {
160 std::unique_lock<std::mutex> lock(mtx);
161 go_produce.wait(lock, [&task_queue]() { return !task_queue.isFull(); });
162 task_queue.enqueue(task);
163 go_work.notify_one();
164 }
165 }
166
167 production_stopped = true;
168 go_work.notify_all();
169 });
170
171 std::vector<std::thread> workers;
172 for (int i = 0; i < num_workers; ++i) {
173 workers.emplace_back([&]() {
174 while (true) {
175 task_t task;
176 {
177 std::unique_lock<std::mutex> lock(mtx);
178 go_work.wait(lock, [&]() { return !task_queue.isEmpty() || production_stopped; });
179 if (task_queue.isEmpty() && production_stopped) {
180 break;
181 }
182 task_queue.dequeue(task);
183 go_produce.notify_one();
184 }
185 ComputingBlockReconstruction<TensorType, MatrixType, ErrorType>(tensor, core_tensor, factor_matrices, error_T, rank, task.block);
186
187 index_t *curr_block_coord = task.block->get_block_coord();
188 for (int axis = 0; axis < tensor->order; ++axis) {
189 --factor_matrices_ref_counter[axis][curr_block_coord[axis]];
190 if (factor_matrices_ref_counter[axis][curr_block_coord[axis]] == 0) {
191 cputucker::deallocate(factor_matrices[axis][curr_block_coord[axis]]);
192 factor_matrices[axis][curr_block_coord[axis]] = nullptr;
193 }
194 }
195
196 // task.block->DestroyBlock();
197 delete task.block;
198 }
199 });
200 }
201
202 producer.join();
203 for (int i = 0; i < num_workers; ++i) {
204 workers[i].join();
205 }
