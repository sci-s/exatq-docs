52template <typename TensorType, typename MatrixType, typename TensorManagerType>
53void ComputingBC(TensorType *tensor, MatrixType **B, MatrixType **C, int curr_factor_id, int rank, TensorManagerType *manager) {
54 using tensor_t = TensorType;
55 using block_t = typename tensor_t::block_t;
56 using index_t = typename tensor_t::index_t;
57 using value_t = typename tensor_t::value_t;
58 using task_t = Task<block_t, value_t>;
59
60 index_t *block_dims = tensor->block_dims;
61 index_t *part_dims = tensor->partition_dims;
62
63 const index_t row_count = block_dims[curr_factor_id];
64 const index_t part_count = part_dims[curr_factor_id];
65 const uint64_t block_count = tensor->block_count;
66 const int num_workers = 4;
67
68 std::atomic<bool> production_stopped{false};
69 std::mutex mtx;
70 std::condition_variable go_produce, go_work;
71
72 CircularQueue<task_t> block_queue(block_count);
73
74 block_t *curr_block = nullptr;
75 value_t *curr_delta = nullptr;
76 task_t task;
77
78 // Initialize B and C
79 int k, l;
80 for (uint64_t part_id = 0; part_id < part_dims[curr_factor_id]; ++part_id) {
81#pragma omp parallel for schedule(static)
82 for (index_t row = 0; row < row_count; ++row) {
83 uint64_t pos_B = row * rank * rank;
84 uint64_t pos_C = row * rank;
85 for (k = 0; k < rank; ++k) {
86 for (l = 0; l < rank; ++l) {
87 B[part_id][pos_B] = 0.0f;
88 if (k == l) {
89 B[part_id][pos_B] = cputucker::constants::kLambda;
90 }
91 ++pos_B;
92 }
93 C[part_id][pos_C] = 0.0f;
94 ++pos_C;
95 }
96 } // !omp parallel
97 } // !part_dims
98
99 std::thread producer([&]() {
100 for (uint64_t block_id = 0; block_id < block_count; ++block_id) {
101 manager->ReadBlockFromFile(curr_block, block_id);
102 curr_delta = cputucker::allocate<value_t>(tensor->get_max_nnz_count_in_block() * rank);
103 manager->ReadDeltaFromFile(curr_delta, tensor, curr_block, rank);
104
105 task = {curr_block, curr_delta};
106
107 {
108 std::unique_lock<std::mutex> lock(mtx);
109 go_produce.wait(lock, [&block_queue]() { return !block_queue.isFull(); });
110 block_queue.enqueue(task);
111 go_work.notify_one();
112 }
113 }
114
115 production_stopped = true;
116 go_work.notify_all();
117 });
118
119 std::vector<std::thread> workers;
120 std::vector<MatrixType **> Local_Bs(num_workers);
121 std::vector<MatrixType **> Local_Cs(num_workers);
122 for (int tid = 0; tid < num_workers; ++tid) {
123 Local_Bs[tid] = cputucker::allocate<MatrixType *>(part_count);
124 Local_Cs[tid] = cputucker::allocate<MatrixType *>(part_count);
125
126 for (uint64_t part_id = 0; part_id < part_dims[curr_factor_id]; ++part_id) {
127 Local_Bs[tid][part_id] = cputucker::allocate<MatrixType>(row_count * rank * rank);
128 Local_Cs[tid][part_id] = cputucker::allocate<MatrixType>(row_count * rank);
129 }
130 workers.emplace_back([&, tid]() {
131 while (true) {
132 task_t task;
133 {
134 std::unique_lock<std::mutex> lock(mtx);
135 go_work.wait(lock, [&]() { return !block_queue.isEmpty() || production_stopped; });
136 if (block_queue.isEmpty() && production_stopped) {
137 break;
138 }
139 block_queue.dequeue(task);
140 go_produce.notify_one();
141 }
142 ComputingBlockBC<TensorType, MatrixType, block_t, value_t>(tensor, Local_Bs[tid], Local_Cs[tid], curr_factor_id, rank, task.block, task.delta);
143 delete task.block;
144 delete task.delta;
145 }
146 });
147 }
148
149 producer.join();
150 for (int i = 0; i < num_workers; ++i) {
151 workers[i].join();
152 }
153
154 for (int i = 0; i < num_workers; ++i) {
155 for (uint64_t part_id = 0; part_id < part_count; ++part_id) {
156 for (index_t row = 0; row < row_count; ++row) {
157 uint64_t pos_B = row * rank * rank;
158 uint64_t pos_C = row * rank;
159 for (int k = 0; k < rank; ++k) {
160 for (int l = 0; l < rank; ++l) {
161 B[part_id][pos_B] += Local_Bs[i][part_id][pos_B];
162 ++pos_B;
163 }
164 C[part_id][pos_C] += Local_Cs[i][part_id][pos_C];
165 ++pos_C;
166 }
167 }
168 }
169 }
170}
